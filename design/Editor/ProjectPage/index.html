use std::error::Error;
use std::fmt;

#[derive(Debug)]
pub struct ValidationError {
    message: String,
}

impl fmt::Display for ValidationError {
    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
        write!(f, "Validation error: {}", self.message)
    }
}

impl Error for ValidationError {}

/// Process and transform data with validation
pub fn process_data(data: Vec<String>) -> Result<Vec<String>, Box<dyn Error>> {
    data.into_iter()
        .map(|s| {
            if s.is_empty() {
                Err(Box::new(ValidationError {
                    message: "Empty string not allowed".to_string(),
                }))
            } else {
                Ok(s.to_uppercase())
            }
        })
        .collect()
}

/// Validate and sanitize input strings
pub fn sanitize_input(input: &str) -> String {
    input
        .trim()
        .replace("\n", " ")
        .replace("\t", " ")
}

/// Calculate similarity score between two strings
pub fn similarity_score(a: &str, b: &str) -> f64 {
    let a_len = a.len() as f64;
    let b_len = b.len() as f64;
    let max_len = a_len.max(b_len);
    
    if max_len == 0.0 {
        return 1.0;
    }
    
    1.0 - (levenshtein_distance(a, b) as f64 / max_len)
}
